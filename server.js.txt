const express = require('express');
const axios = require('axios');
const cron = require('node-cron');
const fs = require('fs');

const app = express();
app.use(express.json());
app.use(express.static('public'));

const DB = './boletos.json';

function lerBoletos() {
  if (!fs.existsSync(DB)) return [];
  return JSON.parse(fs.readFileSync(DB));
}

function salvarBoletos(data) {
  fs.writeFileSync(DB, JSON.stringify(data, null, 2));
}

async function enviarWhatsApp(numero, mensagem) {
  try {
    await axios.post(`https://api.z-api.io/${process.env.ZAPI_ID}/sessions/${process.env.ZAPI_TOKEN}/send-message`, {
      phone: numero,
      message: mensagem
    });
  } catch (e) {
    console.log('Erro:', e.message);
  }
}

async function cobrarBoletos() {
  let hoje = new Date().toLocaleDateString('sv-SE');
  let boletos = lerBoletos();

  for (let b of boletos) {
    if (!b.pago) {
      if (b.vencimento === hoje && !b.cobrado) {
        let msg = `Oi ${b.cliente}, seu boleto vence hoje.`;
        await enviarWhatsApp(b.telefone, msg);
        b.cobrado = true;
      }

      if (b.vencimento < hoje) {
        let msg = `⚠️ ${b.cliente}, seu boleto está em atraso.`;
        await enviarWhatsApp(b.telefone, msg);
      }
    }
  }

  salvarBoletos(boletos);
}

cron.schedule('*/5 * * * *', cobrarBoletos);

app.get('/boletos', (req, res) => {
  res.json(lerBoletos());
});

app.post('/boleto', (req, res) => {
  let boletos = lerBoletos();
  boletos.push({ ...req.body, pago: false, cobrado: false });
  salvarBoletos(boletos);
  res.send("OK");
});

app.post('/pagar/:id', (req, res) => {
  let boletos = lerBoletos();
  boletos[req.params.id].pago = true;
  salvarBoletos(boletos);
  res.send("Pago");
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log('Rodando'));
