<!DOCTYPE html>
<html>
<head>
<title>Sistema de Cobrança</title>
</head>
<body>

<h2>Cadastrar</h2>

<input id="cliente" placeholder="Nome"><br>
<input id="telefone" placeholder="Telefone"><br>
<input id="valor" placeholder="Valor"><br>
<input id="vencimento" type="date"><br>

<button onclick="cadastrar()">Cadastrar</button>

<h2>Boletos</h2>
<div id="lista"></div>

<script>
async function cadastrar() {
  await fetch('/boleto', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      cliente: cliente.value,
      telefone: telefone.value,
      valor: valor.value,
      vencimento: vencimento.value
    })
  });
  carregar();
}

async function pagar(id) {
  await fetch('/pagar/' + id, { method: 'POST' });
  carregar();
}

async function carregar() {
  let res = await fetch('/boletos');
  let data = await res.json();

  lista.innerHTML = '';

  data.forEach((b, i) => {
    lista.innerHTML += `
      <div>
        ${b.cliente} - R$ ${b.valor} - ${b.vencimento}
        ${b.pago ? '✅ Pago' : `<button onclick="pagar(${i})">Pagar</button>`}
      </div>
    `;
  });
}

carregar();
</script>

</body>
</html>
